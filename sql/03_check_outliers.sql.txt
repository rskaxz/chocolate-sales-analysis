-- Statistical distribution of key metrics
SELECT ROUND (AVG (unit_price),2) AS avg_price,
ROUND (MIN(unit_price),2) AS min_price,
ROUND (MAX(unit_price),2) AS max_price,
ROUND (AVG(quantity),2) AS avg_qty,
ROUND (MAX(quantity),2) AS max_qty,
ROUND (AVG(discount*100),1) AS avg_disc
FROM `hayow-481503.chocolate_sales.fact_sales`
